ctrl-b               = toggle side panel
ctrl-j               = toggle terminal/console
ctrl-tab             = tab switcher
ctrl-shift-tab       = go back
ctrl-pageup/pagedown = next/prev tab

ctrl-\               = new vertical tab with current file
ctrl-k-\             = new horizontal tab with current file

ctrl-shift-t         = reopen closed editor
ctrl-num             = open numth tab

ctrl-delete          = delete to the next end of word
ctrl-right arrow     = next word
ctrl-backspace       = delete to the last beggining of word
ctrl-g               = go to line

ctrl-f4              = close editor

ctrl-k ctrl-r        = vscode keyboard shortcut webpage
ctrl-k ctrl-s        = vscode edit shortcuts
